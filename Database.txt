CREATE PROCEDURE [dbo].[Employee_InsertSp]
@id INT ,
@EmpName VARCHAR (100),
 @EmpAge INT ,
 @EmpGender VARCHAR (50),
 @EmpSalary DECIMAL,
 @EmpCountry VARCHAR (50)
AS
BEGIN

 SET NOCOUNT ON; 
  
   IF EXISTS (SELECT 1 FROM EmployeeData WHERE id = @id)
   BEGIN
      UPDATE EmployeeData
	  SET EmpName = @EmpName,
	      EmpAge = @EmpAge,
		  EmpGender = @EmpGender,
		  EmpSalary = @EmpSalary ,
		  EmpCountry = @EmpCountry
	WHERE id = @id
   END

ELSE
	BEGIN
	   INSERT INTO EmployeeData ( EmpName ,  EmpAge , EmpGender ,  EmpSalary ,  EmpCountry)
	   VALUES
      ( @EmpName , @EmpAge , @EmpGender , @EmpSalary , @EmpCountry )
    END
END
GO

CREATE PROCEDURE [dbo].[EmployeeData_SP]
 @ACTION VARCHAR(50),
 @id INT,
 @EmpName VARCHAR (100),
 @EmpAge INT ,
 @EmpGender VARCHAR (50),
 @EmpSalary DECIMAL,
 @EmpCountry VARCHAR (50)
AS
BEGIN
   
   IF( @ACTION = 'GETALL')
   BEGIN
    SELECT * FROM EmployeeData
   END 

   ELSE IF( @ACTION = 'GETBYID' )
   BEGIN
    SELECT * FROM EmployeeData
	WHERE id = @id
   END

   ELSE IF ( @ACTION = 'INSERT' )
   BEGIN
    INSERT INTO EmployeeData ( EmpName ,  EmpAge , EmpGender ,  EmpSalary ,  EmpCountry)
	VALUES
	( @EmpName , @EmpAge , @EmpGender , @EmpSalary , @EmpCountry )
   END


   ELSE IF ( @ACTION = 'UPDATE' )
   BEGIN
     UPDATE EmployeeData
	 SET EmpName = @EmpName,
	     EmpAge = @EmpAge,
		 EmpGender = @EmpGender ,
		 EmpSalary = @EmpSalary,
		 EmpCountry = @EmpCountry
     WHERE id = @id
   END


   ELSE IF ( @ACTION = 'DELETE' )
   BEGIN
     DELETE EmployeeData
	 WHERE id = @id
   END
END
GO
